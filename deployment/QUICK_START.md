# Configure environment variables
az webapp config appsettings set --resource-group mbabot-rg --name mbabot-app --settings \

    DB_NAME="postgres" \
    DB_USER="postgres.xxxxxxxxxxxxx" \
    DB_PASSWORD="your_password" \
    DB_PORT="6543"
